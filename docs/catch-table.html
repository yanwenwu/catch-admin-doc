# 前端表格组件

WARNING

表格组件在一定的程度上可以满足 80% CURD 操作, 如果不满足还是按照常规要求开发

# 基本配置 Props

# search

搜索配置

 {
      type: Array,
      default() {
        return []
      }
},

# filterParmas

搜索参数双向绑定, 搜索必须

{
    type: Object,
      default() {
        return {}
    }
}

# defaultQueryParams

重置时 默认不清除的搜索参数

{
      type: Array,
      default() {
        return []
      }
}

# apiRoute

api 路由地址

{
    type: String,
    default() {
    return ''
    }
}

# dialog

弹窗配置

{
      type: Object,
      default() {
         return {
           title: '新增',
           visible: false,
         }
      }
}

# form

表单配置

form: {
    type: Object,
    default() {
    /**
    return {
        fApi: {},
        value: {},
        rule: {},
    }*/
    return null
    }
}

# hidePagination

是否隐藏分页

 {
      type: Boolean,
      default: false
}

# headers

表格头部配置

{
    type: Array,
    default() {
    return []
    }
}

# tableEvents

表格事件

{
    type: Object,
    default() {
    return {}
    }
}

# 例子

以敏感词页面为例

<template>
  <catch-table
    ref="catchAdmin"
    :headers="table.headers"
    :border="true"
    :search="[
          { type: 'input', name: 'word', placeholder: '请输入敏感词' },
        ]"
    :filterParams="{ word: ''}"
    :form="form"
    :actions="[
      { icon: 'el-icon-plus', text: '新增', class: 'filter-item fr', action: () => {
         this.$refs.catchTable.handleShowDialog()
      }}
    ]"
    :table-events="{
        'selection-change': handleSelectMulti
     }"
    apiRoute="sensitive/word"
  />
</template>
<script>
import CatchTable from '@/components/Catch/Table'
export default {
  components: {
    CatchTable: CatchTable,
  },
  data() {
    return {
      table: {
        headers: [
          {
            type: 'selection',
            width: 50,
          },
          {
            label: '编号',
            prop: 'id',
            width: 80,
          },
          {
            label: '敏感词',
            prop: 'word'
          },
          {
            label: '创建人',
            prop: 'creator'
          },
          {
            label: '创建时间',
            prop: 'created_at'
          },
          {
            label: '更新时间',
            prop: 'created_at'
          },
          {
            label: '操作',
            width: '260',
            isBubble: false,
          }
        ]
      },
      form: {
        fApi: {},
        rule: [
          {
            type: 'input',
            field: 'word',
            title: '敏感词',
            validate: [
              {type: 'string', required: true, message:'请输入敏感词'}
            ]
          },
          {
            type: 'hidden',
            field: 'id',
          }
        ],
      }
    }
  },
  methods: {
    handleSelectMulti(rows) {
      this.$refs.catchTable.handleSelectMulti(rows)
    }
  }
}
</script>
